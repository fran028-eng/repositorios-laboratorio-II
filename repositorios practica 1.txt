repositorios practica 1


# 1. Actualizar repositorios y sistema
sudo apt update && sudo apt upgrade -y

# 2. Ver repositorios en uso
grep ^deb /etc/apt/sources.list /etc/apt/sources.list.d/* 2>/dev/null

# 3. Buscar si bashtop est치 en repositorios
apt search bashtop

# 4. Instalar bashtop desde repositorios si est치 disponible
sudo apt install bashtop -y || echo "Bashtop no est치 en repositorios, instalando manualmente..."

# 5. Si no est치 en repositorios, instalar manualmente
sudo apt install wget unzip make -y
cd /tmp
wget https://github.com/aristocratos/bashtop/archive/refs/heads/master.zip
unzip master.zip
cd bashtop-master
sudo make install

# 6. Probar la herramienta
bashtop

# 7. Desinstalar la herramienta (manual o repos)
sudo apt purge bashtop -y 2>/dev/null || sudo make uninstall

# 8. Limpiar dependencias no usadas
sudo apt autoremove -y
